slideFunct <- function(data, window, step){
  total <- length(data)
  spots <- seq(from=1, to=(total-window), by=step)
  result <- vector(length = length(spots))
  for(i in 1:length(spots)){
    result[i] <- mean(data[spots[i]:(spots[i]+window)])
  }
  return(result)
}

merge_tss <- function(sample_list, gene_name, group_name = ""){
  depth_out = 0
  for(sample_name in sample_list){
    table_in = read.table(paste(sample_name, gene_name, sep = "."), header=F, sep="\t", stringsAsFactors = F)
    colnames(table_in) = c("Chr", "Pos", "Depth")
    depth_out = depth_out + table_in$Depth 
      }
  table_in$Depth = depth_out
  table_in$Sample = group_name
  return(table_in)
  }

merge_tss_m <- function(sample_list, gene_name, group_name = ""){
  depth_out = 0
  for(sample_name in sample_list){
    table_in = read.table(paste(sample_name, gene_name, sep = "."), header=T, sep="\t", stringsAsFactors = F)
    depth_out = depth_out + table_in$Mean.Cov 
      } 
  table_in$Depth = depth_out / length(sample_list)
  table_in$Sample = group_name
  table_in$Relative_Depth = c(rep(0,5), slideFunct(table_in$Depth, 10,1), c(rep(0,5)))
  return(table_in)
  }
